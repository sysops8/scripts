    if (( $(echo "$cpu_usage > $CPU_THRESHOLD" | bc -l) )); then
        log "ALERT" "CPU usage is ${cpu_usage}% (threshold: ${CPU_THRESHOLD}%)"
        return 1
    fi
    log "INFO" "CPU usage: ${cpu_usage}%"
    return 0
}

# –ü—Ä–æ–≤–µ—Ä–∫–∞ —Å–µ—Ä–≤–∏—Å–∞
check_service() {
    local service=$1
    if systemctl is-active --quiet "$service"; then
        log "INFO" "Service $service is running"
        return 0
    else
        log "ALERT" "Service $service is not running"
        return 1
    fi
}

# –°–æ–∑–¥–∞–Ω–∏–µ –±—ç–∫–∞–ø–∞
create_backup() {
    local source=$1
    local dest=$2
    local timestamp=$(date +%Y%m%d_%H%M%S)
    local backup_file="$dest/backup_${timestamp}.tar.gz"
    
    log "INFO" "Starting backup of $source"
    
    if tar -czf "$backup_file" "$source" 2>/dev/null; then
        local size=$(du -h "$backup_file" | cut -f1)
        log "INFO" "Backup completed: $backup_file ($size)"
        md5sum "$backup_file" > "${backup_file}.md5"
        return 0
    else
        log "ERROR" "Backup failed for $source"
        return 1
    fi
}
